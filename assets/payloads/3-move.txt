// it's a custom extension request called "move" with a bunch of arguments

// map ID
// it's paired with the "mp" var in user's variables OR the "mid" var in updates
mid (int) = int

// direction
d (int) = (
	up = 1
	down = 3
	left = 4
	right = 2
)

// it's the amount of step since you spawn in-game
id (int) = 1..n

// same thing as id, but -1
step (int) = id - 1

// the position you wanna move to
tx, ty (int, int) = int, int

// your current position
x, y (int, int) = int, int

// it's the z-index of the of Vec3D. It's always 0. You can find the value in the user's variable called "tz"
layer (int) = 0

sm (bool) = true / false

// you glitch to any position in a map by simply calling a move to that position, it ignores walls
// you cannot teleport though. Your character will "slide" to the position, so everyone can see you... unless you don't spawn
// You cannot teleport to another map unless you are standing on the expected exit door for that map and provide the right x,y position on the next map.
// So no teleportation on the same map (aside from the sliding glitch), no teleportation on another map, no matter if it's adjacent or not.
